


<span id= "20173401">查找项目经理超过x人的员工</span>
----------


	SELECT
		cnt,
		Staff_Work_Id
	FROM
		(
			SELECT
				COUNT (pmID) AS cnt,
				Staff_Work_Id
			FROM
				(
					SELECT
						COUNT (p.Project_Mgr_Work_Id) AS pmID,
						h.Staff_Work_Id
					FROM
						PDRC_HumanMap h,
						PDRC_Project p
					WHERE
						h.Project_Id = p.id --AND h.Staff_Work_Id = 'B-29803'
					AND p.Project_Start_Date > '2016-7-1'
					GROUP BY
						h.Staff_Work_Id,
						p.Project_Mgr_Work_Id
				) a
			GROUP BY
				a.Staff_Work_Id
		) b
	WHERE
		b.cnt > 5;
